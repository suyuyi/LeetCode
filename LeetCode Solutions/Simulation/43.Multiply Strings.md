class Solution {
    public String multiply(String num1, String num2) {
        if(num1.equals("0")||num2.equals("0")) return "0";
        String _num1 = new StringBuffer(num1).reverse().toString(),_num2 = new StringBuffer(num2).reverse().toString();
        List<String> middle = new ArrayList<>();
        for(int i=0;i<_num2.length();++i){
            int cnt =_num2.charAt(i)-'0';
            if(cnt==0) continue;
            StringBuffer stringBuffer = new StringBuffer();
            for(int j=0;j<i;++j) stringBuffer.append(0);
            List<String> tmp = new ArrayList<>();
            while(cnt>0){
                tmp.add(_num1);
                cnt--;
            }
            stringBuffer.append(bigNumPlus(tmp));
            middle.add(stringBuffer.toString());
        }
        String res = bigNumPlus(middle);
        return new StringBuffer(res).reverse().toString();
    }
    public String bigNumPlus(List<String> nums){
        StringBuffer stringBuffer = new StringBuffer();
        int maxLen=0;
        for(String s:nums) maxLen = Math.max(maxLen,s.length());
        int offset=0;
        for(int i=0;i<maxLen;++i){
            int middle=0;
            for(String s:nums){
                if(s.length()>i) middle+=s.charAt(i)-'0';
            }
            middle+=offset;
            offset = middle/10;
            middle%=10;
            stringBuffer.append(middle);
        }
        if(offset!=0) stringBuffer.append(offset);
        return stringBuffer.toString();
    }
}
